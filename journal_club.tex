\begin{SWNFrame}{PAPER AT A GLANCE}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.57\textwidth}
\JCBullets{
  \item \textbf{Problem:} detect and localize pipe leaks in water distribution networks using pressure observations.
  \item \textbf{Core idea:} train a conditional deep convolutional GAN to predict the \emph{leak-free} pressure image from a demand image.
  \item \textbf{Anomaly signal:} compare predicted leak-free pressure image with observed pressure image using structural similarity index (SSIM).
  \item \textbf{Validation:} L-Town benchmark, 24 hypothetical single-leak scenarios, plus six one-year real-time leaks from BattLeDIM.
}

\begin{SWNFrame}{WHY THIS PROBLEM MATTERS}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.5\textwidth}
\textbf{Operational context}
\DenseBullets{
  \item SCADA systems increasingly provide pressure and flow measurements in real time.
  \item Leaks create abnormal hydraulic behavior, but pressure anomalies can be small, delayed, or masked by demand uncertainty.
  \item Detection asks: \emph{is there a leak?} Localization asks: \emph{where is the likely leak area?}
  \item Localization is not final pinpointing; it narrows the search before field inspection.
}
\end{column}
\begin{column}{0.46\textwidth}
\textbf{Why pressure-only LD\&L is hard}
\DenseBullets{
  \item Flow data enables mass-balance logic, but pressure meters are cheaper and easier to install.
  \item Pressure is less sensitive to distant leaks.
  \item Demand variation and measurement error can resemble leak effects.
  \item Labeled historical leak data are scarce, unreliable, or expensive to obtain.
}
\end{column}
\end{columns}
\vspace{0.1in}
\textbf{Journal-club framing:} the paper is not just applying a neural net; it reformulates hydraulic anomaly detection as image-to-image translation plus image similarity.
\end{SWNFrame}

\begin{SWNFrame}{WHERE THE PAPER SITS IN THE LITERATURE}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\textbf{Data-driven methods}
\DenseBullets{
  \item Use historical monitoring data directly.
  \item Include statistical control limits, residual models, classifiers, and clustering.
  \item Advantage: less dependence on detailed hydraulic physics.
  \item Weakness: supervised variants need labeled leak data; localization performance is debated.
}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Model-based methods}
\DenseBullets{
  \item Compare measurements against a calibrated hydraulic model.
  \item Include inverse modeling, sensitivity matrices, and model-generated classifiers.
  \item Advantage: physically interpretable and useful when real leaks are rare.
  \item Weakness: calibration and repeated hydraulic simulations can be expensive.
}
\end{column}
\end{columns}
\vspace{0.08in}
\textbf{Claimed contribution:} combine hydraulic simulation for normal-data generation with a fast CDCGAN metamodel and SSIM-based anomaly scoring.
\end{SWNFrame}

\begin{SWNFrame}{PROPOSED PIPELINE}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.57\textwidth}
\PaperAsset[width=\linewidth]{fig1-flowchart.png}

\JCSource
\end{column}
\begin{column}{0.39\textwidth}
\small
\textbf{Four pillars}
\DenseBullets{
  \item Generate leak-free demand-pressure pairs with EPANET while sampling demand uncertainty.
  \item Interpolate point observations to 256 x 256 grayscale spatial images using kriging.
  \item Train a CDCGAN to map demand images to pressure images under no-leak conditions.
  \item Detect and localize leaks by comparing predicted and observed pressure images with SSIM.
}
\textbf{Key concept:} the model learns normal hydraulic structure, then treats deviations from that learned structure as anomalies.
\end{column}
\end{columns}
\end{SWNFrame}

\begin{SWNFrame}{THEORY: SEMI-SUPERVISED ANOMALY DETECTION}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.49\textwidth}
\textbf{Why semi-supervised?}
\DenseBullets{
  \item Real labeled leak examples are limited.
  \item Fire-hydrant experiments are costly.
  \item Exhaustive synthetic leak simulation over all locations and magnitudes can be computationally infeasible.
  \item Normal, leak-free data are easier to produce or collect.
}
\end{column}
\begin{column}{0.47\textwidth}
\textbf{How the anomaly boundary is built}
\DenseBullets{
  \item Train on normal demand-pressure behavior.
  \item Given a new demand image, generate the expected normal pressure image.
  \item Compare expected normal pressure with observed pressure.
  \item Low similarity means the observation is outside the learned normal region.
}
\end{column}
\end{columns}
\vspace{0.07in}
\textbf{Important assumption:} points close in learned feature/image space are more likely to share the same normal/leak label.
\end{SWNFrame}

\begin{SWNFrame}{MODEL: PIX2PIX-STYLE CDCGAN}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.54\textwidth}
\PaperAsset[width=\linewidth]{fig2-architecture.png}

\JCSource
\end{column}
\begin{column}{0.42\textwidth}
\small
\textbf{Generator}
\DenseBullets{
  \item Modified U-Net encoder-decoder CNN.
  \item Encoder learns increasingly abstract low-resolution features.
  \item Decoder upsamples back to the pressure-image resolution.
  \item Skip connections preserve spatial information.
}
\textbf{Discriminator}
\DenseBullets{
  \item PatchGAN classifier over overlapping 70 x 70 patches.
  \item Objective combines adversarial loss with L1 image reconstruction loss.
}
\end{column}
\end{columns}
\end{SWNFrame}

\begin{SWNFrame}{DETECTION AND LOCALIZATION LOGIC}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.49\textwidth}
\textbf{Image comparison}
\[
SSIM(x,y)=l(x,y)c(x,y)s(x,y)
\]
\DenseBullets{
  \item SSIM compares luminance, contrast, and structure.
  \item $SSIM_o$: score over the whole pressure image for leak detection.
  \item $SSIM_l$: local score map for leak localization.
  \item The paper reports SSIM as a percentage-style score.
}
\end{column}
\begin{column}{0.47\textwidth}
\textbf{Decision rule}
\DenseBullets{
  \item Estimate no-leak mean and standard deviation of $SSIM_o$.
  \item Hourly detection threshold: mean minus three standard deviations.
  \item Alarm only after five consecutive time steps below threshold.
  \item If alarmed, leak location is the minimum-$SSIM_l$ region.
}
\end{column}
\end{columns}
\vspace{0.07in}
\textbf{Tradeoff:} conservative consecutive-step logic reduces false alarms, but it can delay detection of slow/small leaks.
\end{SWNFrame}

\begin{SWNFrame}{TRAINING DATA AND IMAGE CONSTRUCTION}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.52\textwidth}
\PaperAsset[width=\linewidth]{fig4-maps.png}

\JCSource
\end{column}
\begin{column}{0.44\textwidth}
\textbf{Data generation}
\DenseBullets{
  \item Demand at node $i$, time $t$: normal distribution with mean equal to estimated demand and standard deviation $0.1$ times the mean.
  \item Latin hypercube sampling creates plausible demand and pipe-parameter variation.
  \item 365 daily demand patterns at 5-minute resolution.
  \item 105,120 demand-pressure image pairs; 80\% training, 20\% testing.
}
\textbf{Images}
\DenseBullets{
  \item Kriging interpolation with Gaussian variogram.
  \item Grayscale 256 x 256 images scaled from 0 to 255.
  \item Pressure scaled separately in four pressure zones.
}
\end{column}
\end{columns}
\end{SWNFrame}

\textbf{Two trained models}
\DenseBullets{
  \item CDCGAN780: pressure images from 780 observation points.
  \item CDCGAN33: pressure images from the 33 sensor locations.
  \item Both trained for 30,000 epochs.
}
\textbf{Hyperparameters}
\DenseBullets{
  \item Adam optimizer, learning rate $2 \times 10^{-4}$.
  \item First-moment decay 0.5, batch normalization.
  \item L1 loss weight 100.
  \item Normal weight initialization.
}
\textbf{Validation}
\DenseBullets{
  \item Average MAPE: 1.42\% for CDCGAN780.
  \item Average MAPE: 0.67\% for CDCGAN33.

\textbf{Scenario design}
\DenseBullets{
  \item 24 single-leak simulations.
  \item 8 leak locations across four pressure intervals.
  \item Three leak areas: 0.0005, 0.005, and 0.05 m$^2$.
  \item Leak starts at midnight on day 5 and ends after 24 hours.
  \item Leak area grows stepwise in 10 equal steps.
}
\textbf{Hydraulic relation}
\[
q_L=C_L A_L \sqrt{2gh_L}
\]
\DenseBullets{
  \item $C_L=0.75$ by default.
  \item Each scenario creates 2016 image pairs.
}
\end{column}

\begin{column}{0.44\textwidth}
\textbf{Detection metrics}
\[
TPR=\frac{TP}{TP+FN},\quad
TNR=\frac{TN}{TN+FP},\quad
ACC=\frac{TP+TN}{TP+TN+FP+FN}
\]
\DenseBullets{
  \item TPR near 1 for $A_L=0.05$ m$^2$.
  \item TPR drops strongly as leak area decreases and approaches zero for $A_L=0.0005$ m$^2$.
  \item TNR remains generally high.
  \item Average ACC: 0.91 for 780 OPs and 0.90 for 33 OPs.
}
\textbf{Takeaway:} sensor sparsity barely reduced average accuracy in these tests.
\end{column}

\begin{SWNFrame}{WHAT THE PAPER CONTRIBUTES}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}
\textbf{Technical contribution}
\DenseBullets{
  \item Recasts LD\&L as image-to-image translation plus image similarity.
  \item Uses only normal/leak-free training data for the CDCGAN.
  \item Employs SSIM both globally for detection and locally for localization.
  \item Demonstrates feasibility on a realistic benchmark with sparse sensors.
}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Practical promise}
\DenseBullets{
  \item Fast inference compared with repeated hydraulic model simulation.
  \item Tolerates uncertainty better than a direct residual threshold when the normal manifold is learned well.
  \item Image representation naturally handles spatially distributed pressure information.
  \item Similar performance with 33 OPs and 780 OPs in hypothetical scenarios is operationally attractive.
}
\end{column}
\end{columns}
\vspace{0.08in}
\textbf{Main message:} the method is strongest when leaks are large enough to perturb pressure images before the event ends.
\end{SWNFrame}

\begin{SWNFrame}{CRITICAL DISCUSSION}
\small
\begin{columns}[T,onlytextwidth]
\begin{column}{0.49\textwidth}
\textbf{Assumptions and limitations}
\DenseBullets{
  \item L-Town assumes a known network, no initial leaks, fixed configuration, and no time-varying pipe deterioration.
  \item Demand uncertainty and measurement error are modeled as normal distributions.
  \item Sensor data are assumed accurate, synchronized, and rounded only to two decimals.
  \item Real-time events were small; detection happened close to peak leak rate.
  \item Localization for real-time leaks remained coarse: 160 to 185 m.
}
\end{column}
\begin{column}{0.47\textwidth}
\textbf{Questions for journal club}
\DenseBullets{
  \item Does a GAN add value beyond a deterministic U-Net regression model?
  \item How robust is the method to sensor faults, missing sensors, valve changes, and pump schedule changes?
  \item Is SSIM the best anomaly score for hydraulic images, or would learned residual maps perform better?
  \item How should thresholds be tuned for utilities with different false-alarm costs?
}
\end{column}
\end{columns}
\end{SWNFrame}
